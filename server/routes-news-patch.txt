  // Tech News Feed
  app.get("/api/news", async (_req, res) => {
    try {
      const { getTechNews } = await import("./techNewsFeed");
      res.set("Cache-Control", "public, max-age=1800");
      res.json(await getTechNews());
    } catch (error: any) {
      res.status(400).json(formatErrorResponse(error));
    }
  });
